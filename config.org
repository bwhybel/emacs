* Go mode
#+BEGIN_SRC emacs-lisp
  (defun bind-C-c-C-r-to-remove-unused-imports-in-go ()
    "Function to bind 'C-c C-r' to remove unused imports in go"
    (interactive)
    (local-set-key (kbd "C-c C-r") 'go-remove-unused-imports))

  (defun bind-C-c-C-g-to-go-to-imports-in-go ()
    "Function to bind 'C-c C-g' to move pointer to import list"
    (interactive)
    (local-set-key (kbd "C-c C-g") 'go-goto-imports))

  (use-package go-mode
    :ensure t
    :config
    (add-hook 'before-save-hook 'gofmt-before-save)
    (add-hook 'go-mode-hook 'bind-C-c-C-r-to-remove-unused-imports-in-go)
    (add-hook 'go-mode-hook 'bind-C-c-C-g-to-go-to-imports-in-go))
#+END_SRC
